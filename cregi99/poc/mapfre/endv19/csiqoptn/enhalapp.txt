         SPACE ,
         ENHAAPFX DSECT=&DSECT,                                        X
               VER=ALAPP_VER,          CONTROL BLOCK VERSION           X
               EYE='LAPQ',             CONTROL BLOCK EYE-CATCHER       X
               PREFIX=ALAPP_RQ
         SPACE ,
ALAPP_RQ_DATA     DS X                 BEGINNING OF DATA AREA
         ORG      ALAPP_RQ_DATA
ALAPP_RQ_PKGID    DC CL16' '           PACKAGE ID
*                 RESERVED AREA
ALAPP_RQ_PROMHIST DC CL1' '            RETURN PROMOTION HISTORY OPTION
ALAPP_RQ_RESERVE  DC CL3' '            ** RESERVED **
                  DS 0X
ALAPP_RQ_X        DS 0F                LABEL NEEDED FOR CHDR GEN
         SPACE ,
ALAPP_RQ_VER#1_DATADSLN EQU *-ALAPP_RQ_DATA
ALAPP_RQ_VER#1_DSLN     EQU *-ALAPP_RQ
         SPACE ,

         SPACE ,

ALAPP_RQ_VER#2_DSLN     EQU *-ALAPP_RQ
         SPACE ,
ALAPP_RQ_DATADSLN EQU *-ALAPP_RQ_DATA
ALAPP_RQ_DSLN     EQU *-ALAPP_RQ
         ORG      ALAPP_RQ_DATA
ALAPP_RQ_DATAAREA DS XL(ALAPP_RQ_DATADSLN)     DATA AREA
         SPACE 3

*
*        LIST PACKAGE APPROVERS - RESPONSE BLOCK
*
***********************************************************************
         SPACE ,
         ENHAAPFX DSECT=&DSECT,                                        X
               VER=ALAPP_VER,          KEEP VERSION NUMBERS IN SYNC    X
               EYE='LAPS',             CONTROL BLOCK EYE-CATCHER       X
               PREFIX=ALAPP_RS
         SPACE ,
ALAPP_RS_DATA     DS X                 BEGINNING OF DATA AREA
         ORG      ALAPP_RS_DATA
*           **    PACKAGE APPROVER BASE INFORMATION
ALAPP_RS_PKGID    DS CL16              PACKAGE ID
ALAPP_RS_GRPNM    DS CL16              APPROVER GROUP NAME
ALAPP_RS_ENVNM    DS CL8               APPROVER GROUP'S ENV
ALAPP_RS_OUNIQ    DS FL8               APP GRP OVERFLOW UNIQUENESS
ALAPP_RS_SEQNO    DS F                 RECORD SEQUENCE NUMBER
ALAPP_RS_SITE     DS CL1               SITE ID
ALAPP_RS_APFLG    DS CL1               OVERALL APPROVAL STATUS
ALAPP_RS_APFLG$   EQU C' '                APPROVAL PENDING
ALAPP_RS_APFLG$A  EQU C'A'                PACKAGE HAS BEEN APPROVED
ALAPP_RS_APFLG$D  EQU C'D'                PACKAGE HAS BEEN DENIED
ALAPP_RS_GRPTYP   DS CL1               APPROVER GROUP TYPE
ALAPP_RS_GRPTYP$S EQU C'S'                STANDARD APPROVER GROUP
ALAPP_RS_GRPTYP$E EQU C'E'                EXTERNAL APPROVER GROUP
ALAPP_RS_ADISQ    DS CL1               APPROVER DISQUALIFICATION (Y/N)
ALAPP_RS_QUORUM   DS H                 QUORUM COUNT
ALAPP_RS_NOAPPR   DS H                 NBR OF APPROVERS IN THE GROUP
ALAPP_RS_NOAP$MX  EQU 16               MAX NUMBER OF APPROVERS
*
ALAPP_RS_RUD      DS CL7               LAST RECORD UPDATE DATE
ALAPP_RS_RUT      DS CL5               LAST RECORD UPDATE TIME
ALAPP_RS_RUU      DS CL8               USERID ASSOCIATED WITH LAST UPD
*
***  THE FOLLOWING FIELDS OCCUR ALAPP_RS_NOAPPR TIMES
ALAPP_RS_APID     DS CL8               USERID OF APPROVER
ALAPP_RS_APD      DS CL7               APPROVAL/DENIAL DATE (DDMMMYY)
ALAPP_RS_APT      DS CL5               APPROVAL/DENIAL TIME (HH:MM)
ALAPP_RS_APFLAG   DS CL8               APPROVAL STATUS
*                 EQU  C'APPROVED'        APPROVED BY USER
*                 EQU  C'DENIED  '        DENIED BY USER
*                 EQU  C'DISQUAL '        APPROVER DISQUALIFIED
*                 EQU  C'        '        PENDING
ALAPP_RS_APREQD   DS CL8               APPROVAL REQUIRED INDICATOR
*                 EQU  C'REQUIRED'        APPROVAL REQUIRED
*                 EQU  C'        '        APPROVAL OPTIONAL
ALAPP_RS_ENTY$LN  EQU *-ALAPP_RS_APID  LENGTH OF ONE ENTRY
                  DS    (ALAPP_RS_NOAP$MX-1)XL(ALAPP_RS_ENTY$LN)
                  DS 0X
ALAPP_RS_X        DS 0F                   LABEL NEEDED FOR CHDR GEN
         SPACE ,
ALAPP_RS_VER#1_DATADSLN EQU *-ALAPP_RS_DATAAREA
ALAPP_RS_VER#1_DSLN     EQU *-ALAPP_RS
         SPACE ,
.* ADD VERSION 2 FIELDS AFTER THIS POINT
ALAPP_RS_IPKGID   DS CL16              PROMOTION INTERNAL PACKAGE ID
ALAPP_RS_PKGVER   DS H                 PROMOTION PACKAGE VERSION
                  DS CL22              * RESERVED *
*                                      LABEL NEEDED FOR CHDR GEN
ALAPP_RS_X2       DS 0F                ENSURE ALIGNMENT
         SPACE ,
ALAPP_RS_VER#2_DATADSLN EQU *-ALAPP_RS_DATAAREA
ALAPP_RS_VER#2_DSLN     EQU *-ALAPP_RS
         SPACE ,
ALAPP_RS_DATADSLN EQU *-ALAPP_RS_DATAAREA
ALAPP_RS_DSLN     EQU *-ALAPP_RS
         ORG      ALAPP_RS_DATA
