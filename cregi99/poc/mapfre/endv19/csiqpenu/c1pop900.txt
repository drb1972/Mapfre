)ATTR DEFAULT(%+_)
      /* % TYPE(TEXT) INTENS(HIGH)        DEFAULT DISPLAY FOR  */
      /* + TYPE(TEXT) INTENS(LOW)         INFORMATION ONLY     */
      /* _ TYPE(INPUT) INTENS(HIGH) CAPS(ON) JUST (LEFT)       */
   | TYPE(OUTPUT) INTENS(LOW) CAPS(OFF)
   ! TYPE(OUTPUT) INTENS(HIGH) CAPS(OFF)
   @ TYPE(INPUT) INTENS(HIGH) CAPS(OFF) JUST (LEFT)
)BODY WINDOW(70,9) LMSG(LMESS)
+
!LMESS                                        _ZAMT%SCROLL ===>_AMT +
+FROM ELEMENT:
+ %===>|ENTEN255                                           |VNTIND1
+
+TO ELEMENT:
+ %===>@ENT2E255                                           |VNTIND2
+  UPPERCASE%===>_Z+(Y/N)
)INIT
 .HELP = CITNDVRO
 .ZVARS = '(VNTUPPER )'
 &ZWINTTL = 'ENTER "TO" ELEMENT NAME'
 &SWITCH = 'OFF'
 IF (&VNTENLNG > 10)
   &SWITCH = 'ON '
 &SAVSET = ZPFCTL
 &ZPFCTL = OFF
 VPUT(ZPFCTL) PROFILE
 &SAVZAMT = ZAMT
 &ZAMT = '    '
 VPUT(ZAMT) PROFILE
)PROC
 VPUT (AMT) PROFILE
 &ZPFCTL = SAVSET
 VPUT(ZPFCTL) PROFILE
 &ZAMT = SAVZAMT
 VPUT(ZAMT) PROFILE
)FIELD
 FIELD(ENTEN255) IND(VNTIND1) SCROLL(SWITCH)
 FIELD(ENT2E255) IND(VNTIND2) SCROLL(ON)
)END
