         MACRO
         $CPYRITE &LENGTH=FULL,&YEARS=(1986,2013)
         GBLB  &CPYRITE
         LCLA  &YEARCNT,&YRWORK      NUMBER OF RELEASE YEARS AND CURSOR
         LCLC  &YRSTRNG              STRING OF RELEASE YEARS
         AIF   (&CPYRITE).MEXIT
&CPYRITE SETB  1
&YEARCNT SETA  N'&YEARS
&YRWORK  SETA  1
.GENYEAR ANOP
         AIF   (&YRWORK GT &YEARCNT).GENRITE
&YRSTRNG SETC  '&YRSTRNG'.'&YEARS(&YRWORK) '
&YRWORK  SETA  &YRWORK+1
         AGO   .GENYEAR
.GENRITE ANOP
         DC    C' Copyright (C) 1986-2013 CA. All Rights Reserved.'
         DS    0H
.MEXIT   ANOP
         MEND
