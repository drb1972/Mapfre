         SPACE ,
&LABEL   ENHAAPFX DSECT=&DSECT,                                        X
               PREFIX=AACTL
         SPACE ,

         ORG    AACTL_DATA

AACTL_LIST_DDN  DC  CL8' '             RETURN LIST DDNAME
AACTL_HI_MSGID  DC  CL8' '             HIGHEST MSG ID ISSUED

AACTL_RESERVE2  DC  CL1' '             RESERVED

                DS 0D

         SPACE ,

                DS 0D
AACTL_VER#2_DATADSLN EQU *-AACTL_DATA
AACTL_VER#2_DSLN     EQU *-AACTL
         SPACE ,
.* ADD VERSION 3 FIELDS AFTER THIS POINT
         SPACE ,
AACTL_DATADSLN  EQU *-AACTL_DATA
AACTL_DSLN      EQU *-AACTL
         ORG    AACTL_DATA
