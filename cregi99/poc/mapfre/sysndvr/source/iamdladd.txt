%DDNAME=SUCCESS,COND=SUCCESS
  ADD  ELEMENT &C1MEMBER
    FROM DSNAME '&C1FDSN'
    TO  ENVIRONMENT  &ENV
        SYSTEM       &SYS
        SUBSYSTEM    &SBS
        TYPE         &C1TYPE
    OPTIONS CCID '&CCID'
        COMMENT '&COMMENT'
        PROCESSOR GROUP &C1PRGRP
        BYPASS GENERATE
        UPDATE IF PRESENT
  .
